<script>
  const barDataSets = [
    [40], 
    [40, 55], 
    [35, 55, 50], 
    [45, 60, 50, 40], 
    [40, 65, 55, 35, 45], 
    [50, 60, 55, 40, 45, 35]
  ];

  const pieDataSets = [
    [40, 30, 30],
    [50, 20, 30],
    [25, 50, 25],
    [40, 40, 20],
    [35, 40, 25],
    [30, 40, 30],
  ];

  const barColors = '#a57c2b';
  const pieColors = ['#a57c2b', '#deb887', '#f5deb3'];

  function createBarChart(ctx, data) {{
    return new Chart(ctx, {{
      type: 'bar',
      data: {{
        labels: data.map((_, i) => `Bar ${{i+1}}`),
        datasets: [{{
          label: 'Progress',
          data: data,
          backgroundColor: barColors,
          borderRadius: 4,
          maxBarThickness: 40,
        }}]
      }},
      options: {{
        responsive: true,
        animation: {{ duration: 800, easing: 'easeOutQuart' }},
        scales: {{
          y: {{
            beginAtZero: true,
            max: 100,
            ticks: {{ stepSize: 20 }}
          }},
          x: {{
            grid: {{ display: false }}
          }}
        }},
        plugins: {{
          legend: {{ display: false }},
          tooltip: {{
            callbacks: {{
              label: ctx => ctx.parsed.y + '%'
            }}
          }}
        }}
      }}
    }});
  }}

  function createPieChart(ctx, data) {{
    return new Chart(ctx, {{
      type: 'pie',
      data: {{
        labels: ['A', 'B', 'C'],
        datasets: [{{
          data: data,
          backgroundColor: pieColors,
          hoverOffset: 20
        }}]
      }},
      options: {{
        responsive: true,
        animation: {{ duration: 1000 }},
        plugins: {{
          legend: {{
            position: 'bottom',
            labels: {{
              color: '#5e412f',
              font: {{ size: 14, weight: 'bold' }}
            }}
          }}
        }}
      }}
    }});
  }}

  window.onload = () => {{
    for(let i=0; i<barDataSets.length; i++) {{
      const barCtx = document.getElementById(`barChart${i}`).getContext('2d');
      createBarChart(barCtx, barDataSets[i]);

      const pieCtx = document.getElementById(`pieChart${i}`).getContext('2d');
      createPieChart(pieCtx, pieDataSets[i]);
    }}
  }}
</script>
